Perusall Article Downloader

Instructions
  1. Open your Perusall article in Chrome/Firefox
  2. Press F12 to open Developer Tools
  3. Click on Console tab
  4. Copy and paste the code below into the console
  5. Wait for the PDF to be generated and downloaded
JavaScript Code 
void (async function() {
    try {
        // Prevent multiple runs
        if (window.downloaderRunning) {
            console.log('Downloader already running...');
            return;
        }
        window.downloaderRunning = true;

        // Load jsPDF if not already loaded
        if (!window.jspdf) {
            console.log('Loading jsPDF library...');
            await new Promise((resolve, reject) => {
                const script = document.createElement('script');
                script.src = 'https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js';
                script.onload = () => {
                    console.log('jsPDF loaded successfully');
                    resolve();
                };
                script.onerror = reject;
                document.head.appendChild(script);
            });
        }

        // Initialize variables
        let len = 0;
        let times = 0;
        let scrollPasses = 0;
        const maxScrollPasses = 3; // Number of full scroll passes
        
        // Helper function to load an image
        const loadImage = (url) => new Promise((resolve, reject) => {
            const img = new Image();
            img.onload = () => resolve(img);
            img.onerror = reject;
            img.src = url;
        });
        
        // Scroll and collect images with multiple passes
        await new Promise((resolve) => {
            const scrollInterval = setInterval(async () => {
                const images = document.querySelectorAll('img.chunk');
                console.log(`Current images: ${images.length}, Last length: ${len}, Times: ${times}, Passes: ${scrollPasses}`);
                
                if (images.length > 0) {
                    images[images.length - 1].scrollIntoView();
                }
                
                if (len < images.length) {
                    len = images.length;
                    times = 0;
                } else {
                    times++;
                    if (times > 3) {
                        // Scroll back to top
                        window.scrollTo(0, 0);
                        
                        scrollPasses++;
                        times = 0;
                        
                        if (scrollPasses >= maxScrollPasses) {
                            clearInterval(scrollInterval);
                            resolve();
                        }
                    }
                }
            }, 3000);
        });

        // Get all images
        const images = Array.from(document.querySelectorAll('img.chunk'));
        console.log(`Total images found: ${images.length}`);
        
        // Group images by their vertical position (6 chunks per group)
        const chunks = [];
        let currentChunks = [];
        
        // Helper function to get page number from URL or data attributes
        const getPageNumber = (img) => {
            // Try URL patterns first
            const patterns = [
                /page=(\d+)/, // Standard pattern
                /page\/(\d+)/, // Alternative pattern
                /p(\d+)\//, // Another pattern
                /page-(\d+)/, // Hyphenated pattern
                /-p(\d+)-/, // Embedded pattern
                /_(\d+)\.[^.]+$/, // Trailing number pattern
                /\/(\d+)\./, // Number before extension
                /chunk[_-](\d+)/, // Chunk number pattern
                /section[_-](\d+)/, // Section number pattern
            ];
            
            // Try each pattern
            for (const pattern of patterns) {
                const match = img.src.match(pattern);
                if (match) {
                    return parseInt(match[1]);
                }
            }
            
            // Try data attributes
            const dataAttrs = ['data-page', 'data-page-number', 'page', 'data-index'];
            for (const attr of dataAttrs) {
                const value = img.getAttribute(attr);
                if (value) {
                    return parseInt(value);
                }
            }
            
            // If no page number found, use position
            return null;
        };

        // Wait for all images to load and get their positions
        const imagePositions = await Promise.all(images.map(async (img) => {
            // Get absolute position
            const rect = img.getBoundingClientRect();
            const absoluteTop = rect.top + window.pageYOffset;
            
            // Get page number
            const pageNum = getPageNumber(img);
            
            return {
                img,
                top: absoluteTop,
                pageNum,
                src: img.src
            };
        }));

        // Sort images by page number first, then by position
        const sortedPositions = imagePositions.sort((a, b) => {
            // If both have page numbers, use those
            if (a.pageNum !== null && b.pageNum !== null) {
                return a.pageNum - b.pageNum;
            }
            // If only one has a page number, prioritize it
            if (a.pageNum !== null) return -1;
            if (b.pageNum !== null) return 1;
            // If neither has a page number, use position
            return a.top - b.top;
        });

        // Remove duplicates by src while preserving order
        const seen = new Set();
        const uniquePositions = sortedPositions.filter(pos => {
            if (seen.has(pos.src)) return false;
            seen.add(pos.src);
            return true;
        });

        console.log(`Unique images found: ${uniquePositions.length}`);
        
        // Load and group images
        for (const pos of uniquePositions) {
            const loadedImg = await loadImage(pos.src);
            currentChunks.push(loadedImg);
            
            if (currentChunks.length === 6) {
                chunks.push(currentChunks);
                currentChunks = [];
            }
            
            // Log the page number if available
            if (pos.pageNum !== null) {
                console.log(`Added image from page ${pos.pageNum}`);
            }
        }
        
        // Add any remaining chunks
        if (currentChunks.length > 0) {
            chunks.push(currentChunks);
        }
        
        console.log(`Organized into ${chunks.length} pages`);
        
        // Get article title from various possible locations
        const title = (
            document.querySelector('h1.p-title')?.textContent?.trim() || // Try Perusall title class
            document.querySelector('.p-title')?.textContent?.trim() || // Try without h1
            document.querySelector('h1')?.textContent?.trim() || // Try any h1
            document.querySelector('title')?.textContent?.trim() || // Try page title
            'Perusall_Article'
        ).replace(/[<>:"/\|?*]/g, '-'); // Remove invalid filename characters
        
        // Create PDF using jsPDF
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF({
            orientation: 'p',
            unit: 'pt',
            format: [595.28, 841.89] // A4 size in points
        });
        
        // Process each page
        chunks.forEach((pageChunks, pageIndex) => {
            if (pageIndex > 0) doc.addPage();
            
            const pageWidth = doc.internal.pageSize.getWidth();
            const pageHeight = doc.internal.pageSize.getHeight();
            
            // Calculate height for each chunk (divide page height by 6)
            const chunkHeight = pageHeight / 6;
            
            // Add each chunk vertically
            pageChunks.forEach((chunk, index) => {
                const y = index * chunkHeight;
                
                // Add image to fit full width and calculated height
                doc.addImage(chunk, 'JPEG', 0, y, pageWidth, chunkHeight);
            });
        });
        
        console.log('Saving PDF...');
        doc.save(title + '.pdf');
        window.downloaderRunning = false;
        
    } catch (error) {
        console.error('Error:', error);
        window.downloaderRunning = false;
    }
})();
