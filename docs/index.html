Perusall Article Downloader

Instructions
  1. Open your Perusall article in Chrome/Firefox
  2. Press F12 to open Developer Tools
  3. Click on Console tab
  4. Copy and paste the code below into the console
  5. Wait for the PDF to be generated and downloaded
JavaScript Code 
void (async function() {
    try {
        // Prevent multiple runs
        if (window.downloaderRunning) {
            console.log('Downloader already running...');
            return;
        }
        window.downloaderRunning = true;

        // Load jsPDF if not already loaded
        if (!window.jspdf) {
            console.log('Loading jsPDF library...');
            await new Promise((resolve, reject) => {
                const script = document.createElement('script');
                script.src = 'https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js';
                script.onload = () => {
                    console.log('jsPDF loaded successfully');
                    resolve();
                };
                script.onerror = reject;
                document.head.appendChild(script);
            });
        }

        // Try to detect total pages from Perusall interface
        const detectTotalPages = () => {
            // Try various selectors that might contain page information
            const selectors = [
                '.p-page-count',
                '.page-count',
                '[data-total-pages]',
                '.total-pages',
                // Add more selectors as needed
            ];

            for (const selector of selectors) {
                const element = document.querySelector(selector);
                if (element) {
                    const text = element.textContent || element.getAttribute('data-total-pages');
                    if (text) {
                        const match = text.match(/(\d+)\s*(?:pages?|$)/i);
                        if (match) {
                            return parseInt(match[1]);
                        }
                    }
                }
            }

            // Try to find page numbers in the navigation
            const pageNumbers = Array.from(document.querySelectorAll('[data-page-number], [data-page]'))
                .map(el => parseInt(el.getAttribute('data-page-number') || el.getAttribute('data-page')))
                .filter(num => !isNaN(num));

            if (pageNumbers.length > 0) {
                return Math.max(...pageNumbers);
            }

            return null;
        };

        // Initialize variables
        const expectedPages = detectTotalPages(); // Try to detect total pages
        if (!expectedPages) {
            throw new Error('Could not detect total pages. Please try again.');
        }
        console.log('Detected total pages:', expectedPages);
        
        let len = 0;
        let noNewImagesCount = 0;
        let maxNoNewImages = 10;
        let lastScrollPosition = 0;
        let samePositionCount = 0;
        let maxSamePosition = 3;
        const expectedImages = expectedPages * 6;
        console.log(`Expecting ${expectedImages} images (${expectedPages} pages * 6 images per page)`);
        
        // Helper function to load an image
        const loadImage = (url) => new Promise((resolve, reject) => {
            const img = new Image();
            img.onload = () => resolve(img);
            img.onerror = reject;
            img.src = url;
        });

        // Helper function to load images for a specific page
        const loadPageImages = async (pageNum) => {
            console.log(`Loading images for page ${pageNum}...`);
            
            // Try to find and click page/section links
            const pageLinks = Array.from(document.querySelectorAll('a, span, div'))
                .filter(el => {
                    const text = el.textContent?.toLowerCase() || '';
                    const href = el.getAttribute('href')?.toLowerCase() || '';
                    return text.includes(`page ${pageNum}`) || 
                           href.includes(`page=${pageNum}`) || 
                           href.includes(`page/${pageNum}`) ||
                           el.className.includes(`page-${pageNum}`);
                });

            // Click each potential page link
            for (const link of pageLinks) {
                try {
                    link.click();
                    await new Promise(r => setTimeout(r, 2000));
                } catch (e) {
                    console.log('Could not click page link:', e);
                }
            }

            // Function to check if we have all images for this page
            const getPageImages = () => {
                return Array.from(document.querySelectorAll('img.chunk'))
                    .filter(img => {
                        const pageMatch = img.src.match(/page[=/]?(\d+)/i);
                        return pageMatch && parseInt(pageMatch[1]) === pageNum;
                    });
            };

            // Try to load images with multiple attempts
            let attempts = 0;
            const maxAttempts = 5;
            let pageImages = getPageImages();

            while (attempts < maxAttempts && pageImages.length < 6) {
                console.log(`Page ${pageNum}: Found ${pageImages.length}/6 images (attempt ${attempts + 1}/${maxAttempts})`);

                // Scroll around the current viewport to trigger lazy loading
                const currentPos = window.pageYOffset;
                for (let offset = -window.innerHeight; offset <= window.innerHeight; offset += 100) {
                    window.scrollTo(0, currentPos + offset);
                    await new Promise(r => setTimeout(r, 500));
                }

                // Try clicking on image containers
                const containers = document.querySelectorAll('.page-container, .section-container, .document-page');
                for (const container of containers) {
                    try {
                        container.click();
                        await new Promise(r => setTimeout(r, 500));
                    } catch (e) {
                        console.log('Could not click container:', e);
                    }
                }

                pageImages = getPageImages();
                if (pageImages.length === 6) break;
                attempts++;
            }

            return pageImages;
        };

        // Load images page by page
        const allImages = [];
        for (let page = 1; page <= expectedPages; page++) {
            const pageImages = await loadPageImages(page);
            allImages.push(...pageImages);
            console.log(`Page ${page}: Found ${pageImages.length} images (Total: ${allImages.length}/${expectedImages})`);
            
            // If we couldn't find any images for this page after multiple attempts, log warning
            if (pageImages.length === 0) {
                console.log(`Warning: Could not find any images for page ${page}`);
            }
        }


        // Use the collected images
        console.log(`Total images found: ${allImages.length} (expecting ${expectedImages} for ${expectedPages} pages)`);
        
        // Group images by their vertical position (6 chunks per group)
        const chunks = [];
        let currentChunks = [];
        
        // Helper function to get page number from URL or data attributes
        const getPageNumber = (img) => {
            // Try URL patterns first
            const patterns = [
                /page=(\d+)/, // Standard pattern
                /page\/(\d+)/, // Alternative pattern
                /p(\d+)\//, // Another pattern
                /page-(\d+)/, // Hyphenated pattern
                /-p(\d+)-/, // Embedded pattern
                /_(\d+)\.[^.]+$/, // Trailing number pattern
                /\/(\d+)\./, // Number before extension
                /chunk[_-](\d+)/, // Chunk number pattern
                /section[_-](\d+)/, // Section number pattern
            ];
            
            // Try each pattern
            for (const pattern of patterns) {
                const match = img.src.match(pattern);
                if (match) {
                    return parseInt(match[1]);
                }
            }
            
            // Try data attributes
            const dataAttrs = ['data-page', 'data-page-number', 'page', 'data-index'];
            for (const attr of dataAttrs) {
                const value = img.getAttribute(attr);
                if (value) {
                    return parseInt(value);
                }
            }
            
            // If no page number found, use position
            return null;
        };

        // Wait for all images to load and get their positions
        const imagePositions = await Promise.all(images.map(async (img) => {
            // Get absolute position
            const rect = img.getBoundingClientRect();
            const absoluteTop = rect.top + window.pageYOffset;
            
            // Get page number
            const pageNum = getPageNumber(img);
            
            return {
                img,
                top: absoluteTop,
                pageNum,
                src: img.src
            };
        }));

        // Sort images by page number first, then by position
        const sortedPositions = imagePositions.sort((a, b) => {
            // If both have page numbers, use those
            if (a.pageNum !== null && b.pageNum !== null) {
                return a.pageNum - b.pageNum;
            }
            // If only one has a page number, prioritize it
            if (a.pageNum !== null) return -1;
            if (b.pageNum !== null) return 1;
            // If neither has a page number, use position
            return a.top - b.top;
        });

        // Remove duplicates by src while preserving order
        const seen = new Set();
        const uniquePositions = sortedPositions.filter(pos => {
            if (seen.has(pos.src)) return false;
            seen.add(pos.src);
            return true;
        });

        console.log(`Unique images found: ${uniquePositions.length}`);
        
        // Load and group images
        for (const pos of uniquePositions) {
            const loadedImg = await loadImage(pos.src);
            currentChunks.push(loadedImg);
            
            if (currentChunks.length === 6) {
                chunks.push(currentChunks);
                currentChunks = [];
            }
            
            // Log the page number if available
            if (pos.pageNum !== null) {
                console.log(`Added image from page ${pos.pageNum}`);
            }
        }
        
        // Add any remaining chunks
        if (currentChunks.length > 0) {
            chunks.push(currentChunks);
        }
        
        console.log(`Organized into ${chunks.length} pages`);
        
        // Get article title from various possible locations
        const title = (
            document.querySelector('h1.p-title')?.textContent?.trim() || // Try Perusall title class
            document.querySelector('.p-title')?.textContent?.trim() || // Try without h1
            document.querySelector('h1')?.textContent?.trim() || // Try any h1
            document.querySelector('title')?.textContent?.trim() || // Try page title
            'Perusall_Article'
        ).replace(/[<>:"/\|?*]/g, '-'); // Remove invalid filename characters
        
        // Create PDF using jsPDF
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF({
            orientation: 'p',
            unit: 'pt',
            format: [595.28, 841.89] // A4 size in points
        });
        
        // Process each page
        chunks.forEach((pageChunks, pageIndex) => {
            if (pageIndex > 0) doc.addPage();
            
            const pageWidth = doc.internal.pageSize.getWidth();
            const pageHeight = doc.internal.pageSize.getHeight();
            
            // Calculate height for each chunk (divide page height by 6)
            const chunkHeight = pageHeight / 6;
            
            // Add each chunk vertically
            pageChunks.forEach((chunk, index) => {
                const y = index * chunkHeight;
                
                // Add image to fit full width and calculated height
                doc.addImage(chunk, 'JPEG', 0, y, pageWidth, chunkHeight);
            });
        });
        
        console.log('Saving PDF...');
        doc.save(title + '.pdf');
        window.downloaderRunning = false;
        
    } catch (error) {
        console.error('Error:', error);
        window.downloaderRunning = false;
    }
})();
